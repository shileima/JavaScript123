一、json字符串转化成json对象 js 方法


var str = '[{"name":"刘涛","age":"34"},{"name":"张良","age":"55"},{"name":"晨晨","age":"12"}]'

console.log(str);

var jsonObj = JSON.parse(str);

console.log(jsonObj);

二、json字符串转化成json对象 jQuery方法


var str = '[{"name":"刘涛","age":"34"},{"name":"张良","age":"55"},{"name":"晨晨","age":"12"}]'

console.log(str);

var jsonObj = $.parseJSON(str);

console.log(jsonObj);

 

三、json对象转化成json字符串 js 方法

var jsonObj = [{"name":"刘涛","age":"34"},{"name":"张良","age":"55"},{"name":"晨晨","age":"12"}];

console.log(JSON.stringify(jsonObj));
