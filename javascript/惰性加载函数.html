        var addEvent = (ele, type, handler) => {
        if (window.addEventListener) {
        alert('addEventListener')
        addEvent = (ele, type, handler) => {
        ele.addEventListener(type, handler)
        }
        } else if (window.attachEvent) {
        alert('attachEvent')
